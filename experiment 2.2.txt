<h2>Product Filter</h2>

<select id="filter" onchange="filterProducts()">
  <option value="all">All</option>
  <option value="electronics">Electronics</option>
  <option value="fashion">Fashion</option>
</select>

<div class="products" id="productList"></div>

<script>
  const products = [
    { name: "Mobile", category: "electronics", price: 15000 },
    { name: "Laptop", category: "electronics", price: 55000 },
    { name: "T-Shirt", category: "fashion", price: 599 },
    { name: "Shoes", category: "fashion", price: 1999 }
  ];

  function display(items) {
    productList.innerHTML = items.map(p =>
      `<div class="card">
         <h4>${p.name}</h4>
         <p class="price">â‚¹${p.price.toFixed(2)}</p>
       </div>`
    ).join("");
  }

  function filterProducts() {
    const value = filter.value;
    const result = value === "all"
      ? products
      : products.filter(p => p.category === value);
    display(result);
  }

  display(products);
</script>
